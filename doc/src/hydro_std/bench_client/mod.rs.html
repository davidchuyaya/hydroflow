16pub struct SerializableHistogramWrapper {
17    pub histogram: Rc<RefCell<Histogram<u64>>>,
18}
19
20impl Serialize for SerializableHistogramWrapper {
21    fn serialize<S: serde::Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
22        let mut vec = Vec::new();
23        V2Serializer::new()
24            .serialize(&self.histogram.borrow(), &mut vec)
25            .unwrap();
26        serializer.serialize_bytes(&vec)
27    }
28}
29impl<'a> Deserialize<'a> for SerializableHistogramWrapper {
30    fn deserialize<D: serde::Deserializer<'a>>(deserializer: D) -> Result<Self, D::Error> {
31        let mut bytes: &[u8] = Deserialize::deserialize(deserializer)?;
32        let mut histogram = Deserializer::new().deserialize(&mut bytes).unwrap();
33        // Allow auto-resizing to prevent error when combining
34        histogram.auto(true);
35        Ok(SerializableHistogramWrapper {
36            histogram: Rc::new(RefCell::new(histogram)),
37        })
38    }
39}
40pub struct BenchResult<'a, Client> {
41    pub latency_histogram: Singleton<Rc<RefCell<Histogram<u64>>>, Cluster<'a, Client>, Unbounded>,
42    pub throughput: Singleton<RollingAverage, Cluster<'a, Client>, Unbounded>,
43}

45/// Benchmarks transactional workloads by concurrently submitting workloads
46/// (up to `num_clients_per_node` per machine), measuring the latency
47/// of each transaction and throughput over the entire workload.
48/// * `workload_generator` - Generates a payload `P` for each virtual client
49/// * `transaction_cycle` - Processes the payloads and returns after processing
50///
51/// # Non-Determinism
52/// This function uses non-deterministic wall-clock windows for measuring throughput.
53pub fn bench_client<'a, Client, Payload>(
54    clients: &Cluster<'a, Client>,
55    workload_generator: impl FnOnce(
56        &Cluster<'a, Client>,
57        Stream<(u32, Option<Payload>), Cluster<'a, Client>, Unbounded, NoOrder>,
58    )
59        -> Stream<(u32, Payload), Cluster<'a, Client>, Unbounded, NoOrder>,
60    transaction_cycle: impl FnOnce(
61        Stream<(u32, Payload), Cluster<'a, Client>, Unbounded>,
62    )
63        -> Stream<(u32, Payload), Cluster<'a, Client>, Unbounded, NoOrder>,
64    num_clients_per_node: usize,
65    nondet_throughput_window: NonDet,
66) -> BenchResult<'a, Client>
67where
68    Payload: Clone,
69{
70    let (c_to_proposers_complete_cycle, c_to_proposers) =
71        clients.forward_ref::<Stream<_, _, _, TotalOrder>>();
72
73    let (c_latencies, c_throughput_batches, c_new_payloads) = sliced! {
74        let mut next_virtual_client = use::state(|l| Optional::from(l.singleton(q!(0u32))));
75        let mut timers = use::state_null::<KeyedSingleton<u32, Instant, _, _>>();
76
77        let transaction_results = use(transaction_cycle(c_to_proposers).into_keyed(), nondet!(
78            /// because the transaction processor is required to handle arbitrary reordering
79            /// across *different* keys, we are safe because delaying a transaction result for a key
80            /// will only affect when the next request for that key is emitted with respect to other keys
81        ));
82
83        // Set up virtual clients - spawn new ones each tick until we reach the limit
84        let new_virtual_client = next_virtual_client.clone();
85        next_virtual_client = new_virtual_client.clone().filter_map(
86            q!(move |virtual_id| {
87                if virtual_id < num_clients_per_node as u32 {
88                    Some(virtual_id + 1)
89                } else {
90                    None
91                }
92            }),
93        );
94
95        let new_virtual_client_stream = new_virtual_client.into_stream();
96
97        let c_new_payloads_on_start = new_virtual_client_stream
98            .clone()
99            .map(q!(|virtual_id| (virtual_id, None)))
100            .into_keyed();
101
102        let c_received_quorum_payloads = transaction_results
103            .map(q!(|payload| Some(payload)));
104
105        // Track statistics - timers for latency measurement
106        let c_new_timers_when_leader_elected =
107            new_virtual_client_stream.map(q!(|virtual_id| (virtual_id, Instant::now()))).into_keyed();
108        let c_updated_timers = c_received_quorum_payloads
109            .clone()
110            .map(q!(|_payload| Instant::now()));
111
112        let c_latencies = timers
113            .clone()
114            .get_many_if_present(c_updated_timers.clone())
115            .values()
116            .map(q!(
117                |(prev_time, curr_time)| curr_time.duration_since(prev_time)
118            ));
119
120        timers = timers // Update timers in tick+1 so we can record differences during this tick (to track latency)
121            .into_keyed_stream()
122            .chain(c_new_timers_when_leader_elected)
123            .chain(c_updated_timers)
124            .reduce(q!(|curr_time, new_time| {
125                if new_time > *curr_time {
126                    *curr_time = new_time;
127                }
128            }, commutative = ManualProof(/* max is commutative */)));
129
130        // Throughput tracking
131        let c_throughput_new_batch = c_received_quorum_payloads
132            .clone()
133            .values()
134            .count();
135
136        let c_new_payloads = c_new_payloads_on_start.chain(c_received_quorum_payloads);
137
138        (c_latencies, c_throughput_new_batch.into_stream(), c_new_payloads)
139    };
140
141    let c_new_payloads = workload_generator(clients, c_new_payloads.entries());
142    c_to_proposers_complete_cycle.complete(c_new_payloads.assume_ordering::<TotalOrder>(nondet!(
143        /// We don't send a new write for the same key until the previous one is committed,
144        /// so this contains only a single write per key, and we don't care about order
145        /// across keys.
146    )));
147
148    let c_latencies = c_latencies.fold(
149        q!(move || Rc::new(RefCell::new(Histogram::<u64>::new(3).unwrap()))),
150        q!(
151            move |latencies, latency| {
152                latencies
153                    .borrow_mut()
154                    .record(latency.as_nanos() as u64)
155                    .unwrap();
156            },
157            commutative = ManualProof(
158                /* adding elements to histogram is commutative */
159            )
160        ),
161    );
162
163    let throughput_with_timers = c_throughput_batches
164        .map(q!(|batch_size| (batch_size, false)))
165        .merge_ordered(
166            clients
167                .source_interval(q!(Duration::from_secs(1)), nondet_throughput_window)
168                .map(q!(|_| (0, true))),
169            nondet_throughput_window,
170        );
171
172    let c_throughput = throughput_with_timers
173        .fold(
174            q!(|| (0, { RollingAverage::new() })),
175            q!(|(total, stats), (batch_size, reset)| {
176                if reset {
177                    if *total > 0 {
178                        stats.add_sample(*total as f64);
179                    }
180
181                    *total = 0;
182                } else {
183                    *total += batch_size;
184                }
185            }),
186        )
187        .map(q!(|(_, stats)| stats));
188
189    BenchResult {
190        latency_histogram: c_latencies,
191        throughput: c_throughput,
192    }
193}

195/// Prints transaction latency and throughput results to stdout,
196/// with percentiles for latency and a confidence interval for throughput.
197pub fn print_bench_results<'a, Client: 'a, Aggregator>(
198    results: BenchResult<'a, Client>,
199    aggregator: &Process<'a, Aggregator>,
200    clients: &Cluster<'a, Client>,
201) {
202    let nondet_client_count = nondet!(/** client count is stable in bench */);
203    let nondet_sampling = nondet!(/** non-deterministic samping only affects logging */);
204    let print_tick = aggregator.tick();
205    let client_members = aggregator.source_cluster_members(clients);
206    let client_count = track_membership(client_members)
207        .snapshot(&print_tick, nondet_client_count)
208        .filter(q!(|b| *b))
209        .key_count();
210
211    let keyed_throughputs = results
212        .throughput
213        .sample_every(q!(Duration::from_millis(1000)), nondet_sampling)
214        .send(aggregator, TCP.bincode());
215
216    let latest_throughputs = keyed_throughputs.reduce(q!(
217        |combined, new| {
218            *combined = new;
219        },
220        idempotent = ManualProof(/* assignment is idempotent */)
221    ));
222
223    let clients_with_throughputs_count = latest_throughputs
224        .clone()
225        .snapshot(&print_tick, nondet_client_count)
226        // Remove throughputs from clients that have yet to actually record process
227        .filter(q!(|throughputs| throughputs.sample_mean() > 0.0))
228        .key_count();
229
230    let waiting_for_clients = client_count
231        .clone()
232        .zip(clients_with_throughputs_count)
233        .filter_map(q!(|(num_clients, num_clients_with_throughput)| {
234            if num_clients > num_clients_with_throughput {
235                Some(num_clients - num_clients_with_throughput)
236            } else {
237                None
238            }
239        }));
240
241    waiting_for_clients
242        .clone()
243        .all_ticks()
244        .sample_every(q!(Duration::from_millis(1000)), nondet_sampling)
245        .assume_retries(nondet!(/** extra logs due to duplicate samples are okay */))
246        .for_each(q!(|num_clients_not_responded| println!(
247            "Awaiting {} clients",
248            num_clients_not_responded
249        )));
250
251    let combined_throughputs = sliced! {
252        let latest_throughput_snapshot = use(latest_throughputs, nondet_sampling);
253        latest_throughput_snapshot
254            .values()
255            .reduce(q!(|combined, new| {
256                combined.add(new);
257            }, commutative = ManualProof(/* rolling average is commutative */)))
258    };
259
260    combined_throughputs
261        .sample_every(q!(Duration::from_millis(1000)), nondet_sampling)
262        .batch(&print_tick, nondet_client_count)
263        .cross_singleton(client_count.clone())
264        .filter_if_none(waiting_for_clients.clone())
265        .all_ticks()
266        .assume_retries(nondet!(/** extra logs due to duplicate samples are okay */))
267        .for_each(q!(move |(throughputs, num_client_machines)| {
268            if throughputs.sample_count() >= 2 {
269                let mean = throughputs.sample_mean() * num_client_machines as f64;
270
271                if let Some((lower, upper)) = throughputs.confidence_interval_99() {
272                    println!(
273                        "Throughput: {:.2} - {:.2} - {:.2} requests/s",
274                        lower * num_client_machines as f64,
275                        mean,
276                        upper * num_client_machines as f64
277                    );
278                }
279            }
280        }));
281
282    let keyed_latencies = results
283        .latency_histogram
284        .sample_every(q!(Duration::from_millis(1000)), nondet_sampling)
285        .map(q!(|latencies| {
286            SerializableHistogramWrapper {
287                histogram: latencies,
288            }
289        }))
290        .send(aggregator, TCP.bincode());
291
292    let most_recent_histograms = keyed_latencies
293        .map(q!(|histogram| histogram.histogram.borrow().clone()))
294        .reduce(q!(
295            |combined, new| {
296                // get the most recent histogram for each client
297                *combined = new;
298            },
299            idempotent = ManualProof(/* assignment is idempotent */)
300        ));
301
302    let combined_latencies = sliced! {
303        let latencies = use(most_recent_histograms, nondet_sampling);
304        latencies
305            .values()
306            .reduce(q!(|combined, new| {
307                combined.add(new).unwrap();
308            }, commutative = ManualProof(/* combining histories is commutative */)))
309    };
310
311    combined_latencies
312        .sample_every(q!(Duration::from_millis(1000)), nondet_sampling)
313        .batch(&print_tick, nondet_client_count)
314        .filter_if_none(waiting_for_clients)
315        .all_ticks()
316        .assume_retries(nondet!(/** extra logs due to duplicate samples are okay */))
317        .for_each(q!(move |latencies| {
318            println!(
319                "Latency p50: {:.3} | p99 {:.3} | p999 {:.3} ms ({:} samples)",
320                Duration::from_nanos(latencies.value_at_quantile(0.5)).as_micros() as f64 / 1000.0,
321                Duration::from_nanos(latencies.value_at_quantile(0.99)).as_micros() as f64 / 1000.0,
322                Duration::from_nanos(latencies.value_at_quantile(0.999)).as_micros() as f64
323                    / 1000.0,
324                latencies.len()
325            );
326        }));
