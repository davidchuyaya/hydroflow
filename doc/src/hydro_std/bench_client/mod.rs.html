16pub struct SerializableHistogramWrapper {
17    pub histogram: Rc<RefCell<Histogram<u64>>>,
18}
19
20impl Serialize for SerializableHistogramWrapper {
21    fn serialize<S: serde::Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
22        let mut vec = Vec::new();
23        V2Serializer::new()
24            .serialize(&self.histogram.borrow(), &mut vec)
25            .unwrap();
26        serializer.serialize_bytes(&vec)
27    }
28}
29impl<'a> Deserialize<'a> for SerializableHistogramWrapper {
30    fn deserialize<D: serde::Deserializer<'a>>(deserializer: D) -> Result<Self, D::Error> {
31        let mut bytes: &[u8] = Deserialize::deserialize(deserializer)?;
32        let mut histogram = Deserializer::new().deserialize(&mut bytes).unwrap();
33        // Allow auto-resizing to prevent error when combining
34        histogram.auto(true);
35        Ok(SerializableHistogramWrapper {
36            histogram: Rc::new(RefCell::new(histogram)),
37        })
38    }
39}
40pub struct BenchResult<'a, Client> {
41    pub latency_histogram: Singleton<Rc<RefCell<Histogram<u64>>>, Cluster<'a, Client>, Unbounded>,
42    pub throughput: Singleton<RollingAverage, Cluster<'a, Client>, Unbounded>,
43}

45/// Benchmarks transactional workloads by concurrently submitting workloads
46/// (up to `num_clients_per_node` per machine), measuring the latency
47/// of each transaction and throughput over the entire workload.
48/// * `workload_generator` - Generates a payload `P` for each virtual client
49/// * `transaction_cycle` - Processes the payloads and returns after processing
50///
51/// # Non-Determinism
52/// This function uses non-deterministic wall-clock windows for measuring throughput.
53pub fn bench_client<'a, Client, Payload>(
54    clients: &Cluster<'a, Client>,
55    workload_generator: impl FnOnce(
56        &Cluster<'a, Client>,
57        Stream<(u32, Option<Payload>), Cluster<'a, Client>, Unbounded, NoOrder>,
58    )
59        -> Stream<(u32, Payload), Cluster<'a, Client>, Unbounded, NoOrder>,
60    transaction_cycle: impl FnOnce(
61        Stream<(u32, Payload), Cluster<'a, Client>, Unbounded>,
62    )
63        -> Stream<(u32, Payload), Cluster<'a, Client>, Unbounded, NoOrder>,
64    num_clients_per_node: usize,
65    nondet_throughput_window: NonDet,
66) -> BenchResult<'a, Client>
67where
68    Payload: Clone,
69{
70    let client_tick = clients.tick();
71
72    // Set up an initial set of payloads on the first tick
73    let start_this_tick = client_tick.optional_first_tick(q!(()));
74
75    let c_new_payloads_on_start = start_this_tick.clone().flat_map_ordered(q!(move |_| (0
76        ..num_clients_per_node as u32)
77        .map(move |virtual_id| { (virtual_id, None) })));
78
79    let (c_to_proposers_complete_cycle, c_to_proposers) =
80        clients.forward_ref::<Stream<_, _, _, TotalOrder>>();
81
82    // Whenever all replicas confirm that a payload was committed, send another payload
83    let c_received_quorum_payloads = transaction_cycle(c_to_proposers)
84        .batch(
85            &client_tick,
86            nondet!(
87                /// because the transaction processor is required to handle arbitrary reordering
88                /// across *different* keys, we are safe because delaying a transaction result for a key
89                /// will only affect when the next request for that key is emitted with respect to other keys
90            ),
91        )
92        .map(q!(|(virtual_id, payload)| (virtual_id, Some(payload))));
93
94    let c_new_payloads = workload_generator(
95        clients,
96        c_new_payloads_on_start
97            .chain(c_received_quorum_payloads.clone())
98            .all_ticks(),
99    );
100    c_to_proposers_complete_cycle.complete(c_new_payloads.assume_ordering::<TotalOrder>(nondet!(
101        /// We don't send a new write for the same key until the previous one is committed,
102        /// so this contains only a single write per key, and we don't care about order
103        /// across keys.
104    )));
105
106    // Track statistics
107    let (c_timers_complete_cycle, c_timers) =
108        client_tick.cycle::<Stream<(usize, Instant), _, _, NoOrder>>();
109    let c_new_timers_when_leader_elected = start_this_tick
110        .map(q!(|_| Instant::now()))
111        .flat_map_ordered(q!(
112            move |now| (0..num_clients_per_node).map(move |virtual_id| (virtual_id, now))
113        ));
114    let c_updated_timers = c_received_quorum_payloads
115        .clone()
116        .map(q!(|(key, _payload)| (key as usize, Instant::now())));
117    let c_new_timers = c_timers
118        .clone() // Update c_timers in tick+1 so we can record differences during this tick (to track latency)
119        .chain(c_new_timers_when_leader_elected)
120        .chain(c_updated_timers.clone())
121        .into_keyed()
122        .reduce_commutative(q!(|curr_time, new_time| {
123            if new_time > *curr_time {
124                *curr_time = new_time;
125            }
126        }))
127        .entries();
128    c_timers_complete_cycle.complete_next_tick(c_new_timers);
129
130    let c_latencies = c_timers
131        .join(c_updated_timers)
132        .map(q!(
133            |(_virtual_id, (prev_time, curr_time))| curr_time.duration_since(prev_time)
134        ))
135        .all_ticks()
136        .fold_commutative(
137            q!(move || Rc::new(RefCell::new(Histogram::<u64>::new(3).unwrap()))),
138            q!(move |latencies, latency| {
139                latencies
140                    .borrow_mut()
141                    .record(latency.as_nanos() as u64)
142                    .unwrap();
143            }),
144        );
145
146    let c_stats_output_timer = clients
147        .source_interval(q!(Duration::from_secs(1)), nondet_throughput_window)
148        .batch(&client_tick, nondet_throughput_window)
149        .first();
150
151    let c_throughput_new_batch = c_received_quorum_payloads
152        .count()
153        .filter_if_none(c_stats_output_timer.clone())
154        .map(q!(|batch_size| (batch_size, false)));
155
156    let c_throughput_reset = c_stats_output_timer.map(q!(|_| (0, true))).defer_tick();
157
158    let c_throughput = c_throughput_new_batch
159        .into_stream()
160        .chain(c_throughput_reset.into_stream())
161        .all_ticks()
162        .fold(
163            q!(|| (0, { RollingAverage::new() })),
164            q!(|(total, stats), (batch_size, reset)| {
165                if reset {
166                    if *total > 0 {
167                        stats.add_sample(*total as f64);
168                    }
169
170                    *total = 0;
171                } else {
172                    *total += batch_size;
173                }
174            }),
175        )
176        .map(q!(|(_, stats)| { stats }));
177
178    BenchResult {
179        latency_histogram: c_latencies,
180        throughput: c_throughput,
181    }
182}

184/// Prints transaction latency and throughput results to stdout,
185/// with percentiles for latency and a confidence interval for throughput.
186pub fn print_bench_results<'a, Client: 'a, Aggregator>(
187    results: BenchResult<'a, Client>,
188    aggregator: &Process<'a, Aggregator>,
189    clients: &Cluster<'a, Client>,
190) {
191    let nondet_client_count = nondet!(/** client count is stable in bench */);
192    let nondet_sampling = nondet!(/** non-deterministic samping only affects logging */);
193    let print_tick = aggregator.tick();
194    let client_members = aggregator.source_cluster_members(clients);
195    let client_count = track_membership(client_members)
196        .key_count()
197        .snapshot(&print_tick, nondet_client_count);
198
199    let keyed_throughputs = results
200        .throughput
201        .sample_every(q!(Duration::from_millis(1000)), nondet_sampling)
202        .send_bincode(aggregator);
203
204    let latest_throughputs = keyed_throughputs
205        .reduce_idempotent(q!(|combined, new| {
206            *combined = new;
207        }))
208        // Remove throughputs from clients that have yet to actually record process
209        .filter(q!(|throughputs| throughputs.sample_mean() > 0.0));
210
211    let clients_with_throughputs_count = latest_throughputs
212        .clone()
213        .key_count()
214        .snapshot(&print_tick, nondet_client_count);
215
216    let waiting_for_clients = client_count
217        .clone()
218        .zip(clients_with_throughputs_count)
219        .filter_map(q!(|(num_clients, num_clients_with_throughput)| {
220            if num_clients > num_clients_with_throughput {
221                Some(num_clients - num_clients_with_throughput)
222            } else {
223                None
224            }
225        }));
226
227    waiting_for_clients
228        .clone()
229        .all_ticks()
230        .sample_every(q!(Duration::from_millis(1000)), nondet_sampling)
231        .assume_retries(nondet!(/** extra logs due to duplicate samples are okay */))
232        .for_each(q!(|num_clients_not_responded| println!(
233            "Awaiting {} clients",
234            num_clients_not_responded
235        )));
236
237    let combined_throughputs = latest_throughputs
238        .snapshot(&aggregator.tick(), nondet_sampling)
239        .values()
240        .reduce_commutative(q!(|combined, new| {
241            combined.add(new);
242        }))
243        .latest();
244
245    combined_throughputs
246        .sample_every(q!(Duration::from_millis(1000)), nondet_sampling)
247        .batch(&print_tick, nondet_client_count)
248        .cross_singleton(client_count.clone())
249        .filter_if_none(waiting_for_clients.clone())
250        .all_ticks()
251        .assume_retries(nondet!(/** extra logs due to duplicate samples are okay */))
252        .for_each(q!(move |(throughputs, num_client_machines)| {
253            if throughputs.sample_count() >= 2 {
254                let mean = throughputs.sample_mean() * num_client_machines as f64;
255
256                if let Some((lower, upper)) = throughputs.confidence_interval_99() {
257                    println!(
258                        "Throughput: {:.2} - {:.2} - {:.2} requests/s",
259                        lower * num_client_machines as f64,
260                        mean,
261                        upper * num_client_machines as f64
262                    );
263                }
264            }
265        }));
266
267    let keyed_latencies = results
268        .latency_histogram
269        .sample_every(q!(Duration::from_millis(1000)), nondet_sampling)
270        .map(q!(|latencies| {
271            SerializableHistogramWrapper {
272                histogram: latencies,
273            }
274        }))
275        .send_bincode(aggregator);
276
277    let combined_latencies = keyed_latencies
278        .map(q!(|histogram| histogram.histogram.borrow().clone()))
279        .reduce_idempotent(q!(|combined, new| {
280            // get the most recent histogram for each client
281            *combined = new;
282        }))
283        .snapshot(&aggregator.tick(), nondet_sampling)
284        .values()
285        .reduce_commutative(q!(|combined, new| {
286            combined.add(new).unwrap();
287        }))
288        .latest();
289
290    combined_latencies
291        .sample_every(q!(Duration::from_millis(1000)), nondet_sampling)
292        .batch(&print_tick, nondet_client_count)
293        .filter_if_none(waiting_for_clients)
294        .all_ticks()
295        .assume_retries(nondet!(/** extra logs due to duplicate samples are okay */))
296        .for_each(q!(move |latencies| {
297            println!(
298                "Latency p50: {:.3} | p99 {:.3} | p999 {:.3} ms ({:} samples)",
299                Duration::from_nanos(latencies.value_at_quantile(0.5)).as_micros() as f64 / 1000.0,
300                Duration::from_nanos(latencies.value_at_quantile(0.99)).as_micros() as f64 / 1000.0,
301                Duration::from_nanos(latencies.value_at_quantile(0.999)).as_micros() as f64
302                    / 1000.0,
303                latencies.len()
304            );
305        }));
306}
