22fn insert_counter_node(node: &mut HydroNode, next_stmt_id: &mut usize, duration: syn::Expr) {
23    match node {
24        HydroNode::Placeholder
25        | HydroNode::Unpersist { .. }
26        | HydroNode::Counter { .. } => {
27            std::panic!("Unexpected {:?} found in insert_counter_node", node.print_root());
28        }
29        HydroNode::Source { metadata, .. }
30        | HydroNode::CycleSource { metadata, .. }
31        | HydroNode::Persist { metadata, .. }
32        | HydroNode::Delta { metadata, .. }
33        | HydroNode::Chain { metadata, .. } // Can technically be derived by summing parent cardinalities
34        | HydroNode::CrossSingleton { metadata, .. }
35        | HydroNode::CrossProduct { metadata, .. } // Can technically be derived by multiplying parent cardinalities
36        | HydroNode::Join { metadata, .. }
37        | HydroNode::ResolveFutures { metadata, .. }
38        | HydroNode::ResolveFuturesOrdered { metadata, .. }
39        | HydroNode::Difference { metadata, .. }
40        | HydroNode::AntiJoin { metadata, .. }
41        | HydroNode::FlatMap { metadata, .. }
42        | HydroNode::Filter { metadata, .. }
43        | HydroNode::FilterMap { metadata, .. }
44        | HydroNode::Unique { metadata, .. }
45        | HydroNode::Scan { metadata, .. }
46        | HydroNode::Fold { metadata, .. } // Output 1 value per tick
47        | HydroNode::Reduce { metadata, .. } // Output 1 value per tick
48        | HydroNode::FoldKeyed { metadata, .. }
49        | HydroNode::ReduceKeyed { metadata, .. }
50        | HydroNode::ReduceKeyedWatermark { metadata, .. }
51        | HydroNode::Network { metadata, .. }
52        | HydroNode::ExternalInput { metadata, .. }
53         => {
54            let metadata = metadata.clone();
55            let node_content = std::mem::replace(node, HydroNode::Placeholder);
56
57            let counter = HydroNode::Counter {
58                tag: next_stmt_id.to_string(),
59                duration: duration.into(),
60                input: Box::new(node_content),
61                metadata: metadata.clone(),
62            };
63
64            // when we emit this IR, the counter will bump the stmt id, so simulate that here
65            *next_stmt_id += 1;
66
67            *node = counter;
68        }
69        HydroNode::Tee { .. } // Do nothing, we will count the parent of the Tee
70        | HydroNode::Map { .. } // Equal to parent cardinality
71        | HydroNode::DeferTick { .. } // Equal to parent cardinality
72        | HydroNode::Enumerate { .. }
73        | HydroNode::Inspect { .. }
74        | HydroNode::Sort { .. }
75         => {}
76    }
77}

79fn insert_counter(ir: &mut [HydroRoot], duration: syn::Expr) {
80    traverse_dfir(
81        ir,
82        |_, _| {},
83        |node, next_stmt_id| {
84            insert_counter_node(node, next_stmt_id, duration.clone());
85        },
86    );
87}

89async fn track_process_usage_cardinality(
90    process: &impl DeployCrateWrapper,
91) -> (UnboundedReceiver<String>, UnboundedReceiver<String>) {
92    (
93        process.stdout_filter(CPU_USAGE_PREFIX).await,
94        process.stdout_filter(COUNTER_PREFIX).await,
95    )
96}

98async fn track_cluster_usage_cardinality(
99    nodes: &DeployResult<'_, HydroDeploy>,
100) -> (
101    HashMap<(LocationId, String, usize), UnboundedReceiver<String>>,
102    HashMap<(LocationId, String, usize), UnboundedReceiver<String>>,
103) {
104    let mut usage_out = HashMap::new();
105    let mut cardinality_out = HashMap::new();
106    for (id, name, cluster) in nodes.get_all_clusters() {
107        for (idx, node) in cluster.members().iter().enumerate() {
108            let (node_usage_out, node_cardinality_out) =
109                track_process_usage_cardinality(node).await;
110            usage_out.insert((id.clone(), name.clone(), idx), node_usage_out);
111            cardinality_out.insert((id.clone(), name.clone(), idx), node_cardinality_out);
112        }
113    }
114    for (id, name, process) in nodes.get_all_processes() {
115        let (process_usage_out, process_cardinality_out) =
116            track_process_usage_cardinality(process).await;
117        usage_out.insert((id.clone(), name.clone(), 0), process_usage_out);
118        cardinality_out.insert((id.clone(), name.clone(), 0), process_cardinality_out);
119    }
120    (usage_out, cardinality_out)
121}

123/// TODO: Return type should be changed to also include Partitioner
124pub async fn deploy_and_analyze<'a>(
125    reusable_hosts: &mut ReusableHosts,
126    deployment: &mut Deployment,
127    builder: FlowBuilder<'a>,
128    clusters: &Vec<(usize, String, usize)>,
129    processes: &Vec<(usize, String)>,
130    exclude_from_decoupling: Vec<String>,
131    num_seconds: Option<usize>,
132) -> (
133    RewriteIrFlowBuilder<'a>,
134    Vec<HydroRoot>,
135    Decoupler,
136    String,
137    usize,
138) {
139    let counter_output_duration = syn::parse_quote!(std::time::Duration::from_secs(1));
140
141    // Rewrite with counter tracking
142    let rewritten_ir_builder = builder.rewritten_ir_builder();
143    let optimized = builder.optimize_with(persist_pullup).optimize_with(|leaf| {
144        insert_counter(leaf, counter_output_duration);
145    });
146    let mut ir = deep_clone(optimized.ir());
147
148    // Insert all clusters & processes
149    let mut deployable = optimized.into_deploy();
150    for (cluster_id, name, num_hosts) in clusters {
151        deployable = deployable.with_cluster_id_name(
152            *cluster_id,
153            name.clone(),
154            reusable_hosts.get_cluster_hosts(deployment, name.clone(), *num_hosts),
155        );
156    }
157    for (process_id, name) in processes {
158        deployable = deployable.with_process_id_name(
159            *process_id,
160            name.clone(),
161            reusable_hosts.get_process_hosts(deployment, name.clone()),
162        );
163    }
164    let nodes = deployable.deploy(deployment);
165    deployment.deploy().await.unwrap();
166
167    let (mut usage_out, mut cardinality_out) = track_cluster_usage_cardinality(&nodes).await;
168
169    // Wait for user to input a newline
170    deployment
171        .start_until(async {
172            if let Some(seconds) = num_seconds {
173                // Wait for some number of seconds
174                tokio::time::sleep(Duration::from_secs(seconds as u64)).await;
175            } else {
176                // Wait for a new line
177                eprintln!("Press enter to stop deployment and analyze results");
178                let _ = tokio::io::AsyncBufReadExt::lines(tokio::io::BufReader::new(
179                    tokio::io::stdin(),
180                ))
181                .next_line()
182                .await
183                .unwrap();
184            }
185        })
186        .await
187        .unwrap();
188
189    let (bottleneck, bottleneck_name, bottleneck_num_nodes) = analyze_cluster_results(
190        &nodes,
191        &mut ir,
192        &mut usage_out,
193        &mut cardinality_out,
194        exclude_from_decoupling,
195    )
196    .await;
197    // Remove HydroNode::Counter (since we don't want to consider decoupling those)
198    remove_counter(&mut ir);
199    // Inject new next_stmt_id into metadata (old ones are invalid after removing the counter)
200    inject_id(&mut ir);
201
202    // Create a mapping from each CycleSink to its corresponding CycleSource
203    let cycle_source_to_sink_input = cycle_source_to_sink_input(&mut ir);
204    let (send_overhead, recv_overhead) = analyze_send_recv_overheads(&mut ir, &bottleneck);
205    let (orig_to_decoupled, decoupled_to_orig, place_on_decoupled) = decouple_analysis(
206        &mut ir,
207        &bottleneck,
208        send_overhead,
209        recv_overhead,
210        &cycle_source_to_sink_input,
211    );
212
213    // TODO: Save decoupling decision to file
214
215    (
216        rewritten_ir_builder,
217        ir,
218        Decoupler {
219            output_to_decoupled_machine_after: orig_to_decoupled,
220            output_to_original_machine_after: decoupled_to_orig,
221            place_on_decoupled_machine: place_on_decoupled,
222            orig_location: bottleneck.clone(),
223            decoupled_location: LocationId::Process(0), // Placeholder, must replace
224        },
225        bottleneck_name,
226        bottleneck_num_nodes,
227    )
228}
