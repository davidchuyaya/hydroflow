13pub fn parse_cpu_usage(measurement: String) -> f64 {
14    let regex = Regex::new(r"Total (\d+\.\d+)%").unwrap();
15    regex
16        .captures_iter(&measurement)
17        .last()
18        .map(|cap| cap[1].parse::<f64>().unwrap())
19        .unwrap_or(0f64)
20}

22/// Returns a map from (operator ID, is network receiver) to percentage of total samples.
23fn parse_perf(file: String) -> HashMap<(usize, bool), f64> {
24    let mut total_samples = 0f64;
25    let mut unidentified_samples = 0f64;
26    let mut samples_per_id = HashMap::new();
27    let operator_regex = Regex::new(r"op_\d+v\d+__(.*?)__(send)?(recv)?(\d+)").unwrap();
28
29    for line in file.lines() {
30        let n_samples_index = line.rfind(' ').unwrap() + 1;
31        let n_samples = &line[n_samples_index..].parse::<f64>().unwrap();
32
33        if let Some(cap) = operator_regex.captures_iter(line).last() {
34            let id = cap[4].parse::<usize>().unwrap();
35            let is_network_recv = cap
36                .get(3)
37                .is_some_and(|direction| direction.as_str() == "recv");
38
39            let dfir_operator_and_samples =
40                samples_per_id.entry((id, is_network_recv)).or_insert(0.0);
41            *dfir_operator_and_samples += n_samples;
42        } else {
43            unidentified_samples += n_samples;
44        }
45        total_samples += n_samples;
46    }
47
48    println!(
49        "Out of {} samples, {} were unidentified, {}%",
50        total_samples,
51        unidentified_samples,
52        unidentified_samples / total_samples * 100.0
53    );
54
55    samples_per_id
56        .iter_mut()
57        .for_each(|(_, samples)| *samples /= total_samples);
58    samples_per_id
59}

61fn inject_perf_root(
62    root: &mut HydroRoot,
63    id_to_usage: &HashMap<(usize, bool), f64>,
64    next_stmt_id: &mut usize,
65) {
66    if let Some(cpu_usage) = id_to_usage.get(&(*next_stmt_id, false)) {
67        root.op_metadata_mut().cpu_usage = Some(*cpu_usage);
68    }
69}

71fn inject_perf_node(
72    node: &mut HydroNode,
73    id_to_usage: &HashMap<(usize, bool), f64>,
74    next_stmt_id: &mut usize,
75) {
76    if let Some(cpu_usage) = id_to_usage.get(&(*next_stmt_id, false)) {
77        node.op_metadata_mut().cpu_usage = Some(*cpu_usage);
78    }
79    // If this is a Network node, separately get receiver CPU usage
80    if let HydroNode::Network { metadata, .. } = node
81        && let Some(cpu_usage) = id_to_usage.get(&(*next_stmt_id, true))
82    {
83        metadata.op.network_recv_cpu_usage = Some(*cpu_usage);
84    }
85}

87pub fn inject_perf(ir: &mut [HydroRoot], folded_data: Vec<u8>) {
88    let id_to_usage = parse_perf(String::from_utf8(folded_data).unwrap());
89    traverse_dfir(
90        ir,
91        |leaf, next_stmt_id| {
92            inject_perf_root(leaf, &id_to_usage, next_stmt_id);
93        },
94        |node, next_stmt_id| {
95            inject_perf_node(node, &id_to_usage, next_stmt_id);
96        },
97    );
98}

100/// Returns (op_id, count)
101pub fn parse_counter_usage(measurement: String) -> (usize, usize) {
102    let regex = Regex::new(r"\((\d+)\): (\d+)").unwrap();
103    let matches = regex.captures_iter(&measurement).last().unwrap();
104    let op_id = matches[1].parse::<usize>().unwrap();
105    let count = matches[2].parse::<usize>().unwrap();
106    (op_id, count)
107}

109fn inject_count_node(
110    node: &mut HydroNode,
111    next_stmt_id: &mut usize,
112    op_to_count: &HashMap<usize, usize>,
113) {
114    if let Some(count) = op_to_count.get(next_stmt_id) {
115        let metadata = node.metadata_mut();
116        metadata.cardinality = Some(*count);
117    } else {
118        match node {
119            HydroNode::Tee { inner ,metadata, .. } => {
120                metadata.cardinality = inner.0.borrow().metadata().cardinality;
121            }
122            | HydroNode::Map { input, metadata, .. } // Equal to parent cardinality
123            | HydroNode::DeferTick { input, metadata, .. } // Equal to parent cardinality
124            | HydroNode::Enumerate { input, metadata, .. }
125            | HydroNode::Inspect { input, metadata, .. }
126            | HydroNode::Sort { input, metadata, .. }
127            | HydroNode::Counter { input, metadata, .. }
128            => {
129                metadata.cardinality = input.metadata().cardinality;
130            }
131            _ => {}
132        }
133    }
134}

146pub async fn analyze_process_results(
147    process: &impl DeployCrateWrapper,
148    ir: &mut [HydroRoot],
149    _node_usage: f64,
150    node_cardinality: &mut UnboundedReceiver<String>,
151) {
152    // TODO: Integrate CPU usage into perf usage stats (so we also consider idle time)
153    if let Some(perf_results) = process.tracing_results().await {
154        // Inject perf usages into metadata
155        inject_perf(ir, perf_results.folded_data);
156
157        // Get cardinality data. Allow later values to overwrite earlier ones
158        let mut op_to_counter = HashMap::new();
159        while let Some(measurement) = node_cardinality.recv().await {
160            let (op_id, count) = parse_counter_usage(measurement);
161            op_to_counter.insert(op_id, count);
162        }
163        inject_count(ir, &op_to_counter);
164    }
165}

167pub async fn analyze_cluster_results(
168    nodes: &DeployResult<'_, HydroDeploy>,
169    ir: &mut [HydroRoot],
170    usage_out: &mut HashMap<(LocationId, String, usize), UnboundedReceiver<String>>,
171    cardinality_out: &mut HashMap<(LocationId, String, usize), UnboundedReceiver<String>>,
172    exclude_from_decoupling: Vec<String>,
173) -> (LocationId, String, usize) {
174    let mut max_usage_cluster_id = None;
175    let mut max_usage_cluster_size = 0;
176    let mut max_usage_cluster_name = String::new();
177    let mut max_usage_overall = 0f64;
178
179    for (id, name, cluster) in nodes.get_all_clusters() {
180        println!("Analyzing cluster {:?}: {}", id, name);
181
182        // Iterate through nodes' usages and keep the max usage one
183        let mut max_usage = None;
184        for (idx, _) in cluster.members().iter().enumerate() {
185            let usage =
186                get_usage(usage_out.get_mut(&(id.clone(), name.clone(), idx)).unwrap()).await;
187            println!("Node {} usage: {}", idx, usage);
188            if let Some((prev_usage, _)) = max_usage {
189                if usage > prev_usage {
190                    max_usage = Some((usage, idx));
191                }
192            } else {
193                max_usage = Some((usage, idx));
194            }
195        }
196
197        if let Some((usage, idx)) = max_usage {
198            // Modify IR with perf & cardinality numbers
199            let node_cardinality = cardinality_out
200                .get_mut(&(id.clone(), name.clone(), idx))
201                .unwrap();
202            analyze_process_results(
203                cluster.members().get(idx).unwrap(),
204                ir,
205                usage,
206                node_cardinality,
207            )
208            .await;
209
210            // Update cluster with max usage
211            if max_usage_overall < usage && !exclude_from_decoupling.contains(&name) {
212                max_usage_cluster_id = Some(id.clone());
213                max_usage_cluster_name = name.clone();
214                max_usage_cluster_size = cluster.members().len();
215                max_usage_overall = usage;
216                println!("The bottleneck is {}", name);
217            }
218        }
219    }
220
221    (
222        max_usage_cluster_id.unwrap(),
223        max_usage_cluster_name,
224        max_usage_cluster_size,
225    )
226}

233fn analyze_overheads_node(
234    node: &mut HydroNode,
235    _next_stmt_id: &mut usize,
236    max_send_overhead: &mut f64,
237    max_recv_overhead: &mut f64,
238    location: &LocationId,
239) {
240    let metadata = node.metadata();
241    let network_type = get_network_type(node, location.root().raw_id());
242    match network_type {
243        Some(NetworkType::Send) | Some(NetworkType::SendRecv) => {
244            if let Some(cpu_usage) = metadata.op.cpu_usage {
245                // Use cardinality from the network's input, not the network itself.
246                // Reason: Cardinality is measured at ONE recipient, but the sender may be sending to MANY machines.
247                if let Some(cardinality) = node.input_metadata().first().unwrap().cardinality {
248                    let overhead = cpu_usage / cardinality as f64;
249
250                    println!("New send overhead: {}", overhead);
251                    if overhead > *max_send_overhead {
252                        *max_send_overhead = overhead;
253                    }
254                }
255            }
256        }
257        _ => {}
258    }
259    match network_type {
260        Some(NetworkType::Recv) | Some(NetworkType::SendRecv) => {
261            if let Some(cardinality) = metadata.cardinality
262                && let Some(cpu_usage) = metadata.op.network_recv_cpu_usage
263            {
264                let overhead = cpu_usage / cardinality as f64;
265
266                println!("New receive overhead: {}", overhead);
267                if overhead > *max_recv_overhead {
268                    *max_recv_overhead = overhead;
269                }
270            }
271        }
272        _ => {}
273    }
274}

276// Track the max of each so we decouple conservatively
277pub fn analyze_send_recv_overheads(ir: &mut [HydroRoot], location: &LocationId) -> (f64, f64) {
278    let mut max_send_overhead = 0.0;
279    let mut max_recv_overhead = 0.0;
280
281    traverse_dfir(
282        ir,
283        |_, _| {},
284        |node, next_stmt_id| {
285            analyze_overheads_node(
286                node,
287                next_stmt_id,
288                &mut max_send_overhead,
289                &mut max_recv_overhead,
290                location,
291            );
292        },
293    );
294
295    if max_send_overhead == 0.0 {
296        println!("Warning: No send overhead found.");
297    }
298    if max_recv_overhead == 0.0 {
299        println!("Warning: No receive overhead found.");
300    }
301
302    (max_send_overhead, max_recv_overhead)
