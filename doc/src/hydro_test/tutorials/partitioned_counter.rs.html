7pub struct CounterServer;
8pub struct CounterShard;

10#[expect(clippy::type_complexity, reason = "output types with orderings")]
11pub fn sharded_counter_service<'a>(
12    leader: &Process<'a, CounterServer>,
13    shard_servers: &Cluster<'a, CounterShard>,
14    increment_requests: KeyedStream<u32, String, Process<'a, CounterServer>>,
15    get_requests: KeyedStream<u32, String, Process<'a, CounterServer>>,
16) -> (
17    KeyedStream<u32, String, Process<'a, CounterServer>, Unbounded, NoOrder>,
18    KeyedStream<u32, (String, usize), Process<'a, CounterServer>, Unbounded, NoOrder>,
19) {
20    let sharded_increment_requests = increment_requests
21        .entries()
22        .map(q!(|(client, key)| {
23            let mut hasher = DefaultHasher::new();
24            key.hash(&mut hasher);
25            (
26                MemberId::from_raw_id(hasher.finish() as u32 % 5),
27                (client, key),
28            )
29        }))
30        .into_keyed()
31        .demux_bincode(shard_servers)
32        .into_keyed();
33
34    let sharded_get_requests = get_requests
35        .entries()
36        .map(q!(|(client, key)| {
37            let mut hasher = DefaultHasher::new();
38            key.hash(&mut hasher);
39            (
40                MemberId::from_raw_id(hasher.finish() as u32 % 5),
41                (client, key),
42            )
43        }))
44        .into_keyed()
45        .demux_bincode(shard_servers)
46        .into_keyed();
47
48    let (sharded_increment_ack, sharded_get_response) = super::keyed_counter::keyed_counter_service(
49        sharded_increment_requests,
50        sharded_get_requests,
51    );
52
53    let increment_ack = sharded_increment_ack
54        .entries()
55        .send_bincode(leader)
56        .values()
57        .into_keyed();
58
59    let get_response = sharded_get_response
60        .entries()
61        .send_bincode(leader)
62        .values()
63        .into_keyed();
64
65    (increment_ack, get_response)
66}

69mod tests {

72    use super::*;

74    #[test]
75    fn test_counter_read_after_write() {
76        let flow = FlowBuilder::new();
77        let process = flow.process();
78        let shards = flow.cluster();
79        let external = flow.external::<()>();
80
81        let (inc_in_port, inc_requests) = process.source_external_bincode(&external);
82        let inc_requests = inc_requests.into_keyed();
83
84        let (get_in_port, get_requests) = process.source_external_bincode(&external);
85        let get_requests = get_requests.into_keyed();
86
87        let (inc_acks, get_responses) =
88            sharded_counter_service(&process, &shards, inc_requests, get_requests);
89
90        let inc_out_port = inc_acks.entries().send_bincode_external(&external);
91        let get_out_port = get_responses.entries().send_bincode_external(&external);
92
93        flow.sim()
94            .with_cluster_size(&shards, 5)
95            .exhaustive(async |mut instance| {
96                let inc_in_port = instance.connect(&inc_in_port);
97                let get_in_port = instance.connect(&get_in_port);
98                let mut inc_out_port = instance.connect(&inc_out_port);
99                let get_out_port = instance.connect(&get_out_port);
100
101                instance.launch();
102
103                inc_in_port.send((1, "abc".to_string()));
104                inc_out_port
105                    .assert_yields_unordered([(1, "abc".to_string())])
106                    .await;
107                get_in_port.send((1, "abc".to_string()));
108                get_out_port
109                    .assert_yields_only_unordered([(1, ("abc".to_string(), 1))])
110                    .await;
111            });
112    }
