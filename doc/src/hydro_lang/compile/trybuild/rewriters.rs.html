4pub struct UseTestModeStaged<'a> {
5    pub crate_name: &'a str,
6}
7
8impl VisitMut for UseTestModeStaged<'_> {
9    fn visit_type_path_mut(&mut self, i: &mut syn::TypePath) {
10        if let Some(first) = i.path.segments.first()
11            && first.ident == self.crate_name
12            && let Some(second) = i.path.segments.get(1)
13            && second.ident == "__staged"
14        {
15            i.path.segments.first_mut().unwrap().ident =
16                syn::Ident::new("crate", Span::call_site());
17        }
18
19        syn::visit_mut::visit_type_path_mut(self, i);
20    }
21
22    fn visit_use_path_mut(&mut self, i: &mut syn::UsePath) {
23        if i.ident == self.crate_name {
24            i.ident = syn::Ident::new("crate", i.ident.span());
25        }
26    }
27
28    fn visit_macro_mut(&mut self, i: &mut syn::Macro) {
29        // Rewrite types inside __maybe_debug__!(Type) invocations.
30        if i.path
31            .segments
32            .last()
33            .is_some_and(|s| s.ident == "__maybe_debug__")
34            && let Ok(mut ty) = i.parse_body::<syn::Type>()
35        {
36            self.visit_type_mut(&mut ty);
37            i.tokens = quote::quote! { #ty };
38        }
39
40        syn::visit_mut::visit_macro_mut(self, i);
41    }
42}
