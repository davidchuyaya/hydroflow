1//! Type definitions for distributed locations, which specify where pieces of a Hydro
2//! program will be executed.
3//!
4//! Hydro is a **global**, **distributed** programming model. This means that the data
5//! and computation in a Hydro program can be spread across multiple machines, data
6//! centers, and even continents. To achieve this, Hydro uses the concept of
7//! **locations** to keep track of _where_ data is located and computation is executed.
8//!
9//! Each live collection type (in [`crate::live_collections`]) has a type parameter `L`
10//! which will always be a type that implements the [`Location`] trait (e.g. [`Process`]
11//! and [`Cluster`]). To create distributed programs, Hydro provides a variety of APIs
12//! to allow live collections to be _moved_ between locations via network send/receive.
13//!
14//! See [the Hydro docs](https://hydro.run/docs/hydro/locations/) for more information.
15

96    fn try_tick(&self) -> Option<Tick<Self>> {
97        if Self::is_top_level() {
98            let next_id = self.flow_state().borrow_mut().next_clock_id;
99            self.flow_state().borrow_mut().next_clock_id += 1;
100            Some(Tick {
101                id: next_id,
102                l: self.clone(),
103            })
104        } else {
105            None
106        }
107    }

263    /// Establishes a server on this location to receive a bidirectional connection from a single
264    /// client, identified by the given `External` handle. Returns a port handle for the external
265    /// process to connect to, a stream of incoming messages, and a handle to send outgoing
266    /// messages.
267    ///
268    /// # Example
269    /// ```rust
270    /// # #[cfg(feature = "deploy")] {
271    /// # use hydro_lang::prelude::*;
272    /// # use hydro_deploy::Deployment;
273    /// # use futures::{SinkExt, StreamExt};
274    /// # tokio_test::block_on(async {
275    /// # use bytes::Bytes;
276    /// # use hydro_lang::location::NetworkHint;
277    /// # use tokio_util::codec::LengthDelimitedCodec;
278    /// # let flow = FlowBuilder::new();
279    /// let node = flow.process::<()>();
280    /// let external = flow.external::<()>();
281    /// let (port, incoming, outgoing) =
282    ///     node.bind_single_client::<_, Bytes, LengthDelimitedCodec>(&external, NetworkHint::Auto);
283    /// outgoing.complete(incoming.map(q!(|data /* : Bytes */| {
284    ///     let mut resp: Vec<u8> = data.into();
285    ///     resp.push(42);
286    ///     resp.into() // : Bytes
287    /// })));
288    ///
289    /// # let mut deployment = Deployment::new();
290    /// let nodes = flow // ... with_process and with_external
291    /// #     .with_process(&node, deployment.Localhost())
292    /// #     .with_external(&external, deployment.Localhost())
293    /// #     .deploy(&mut deployment);
294    ///
295    /// deployment.deploy().await.unwrap();
296    /// deployment.start().await.unwrap();
297    ///
298    /// let (mut external_out, mut external_in) = nodes.connect(port).await;
299    /// external_in.send(vec![1, 2, 3].into()).await.unwrap();
300    /// assert_eq!(
301    ///     external_out.next().await.unwrap().unwrap(),
302    ///     vec![1, 2, 3, 42]
303    /// );
304    /// # });
305    /// # }
306    /// ```
307    #[expect(clippy::type_complexity, reason = "stream markers")]
308    fn bind_single_client<L, T, Codec: Encoder<T> + Decoder>(
309        &self,
310        from: &External<L>,
311        port_hint: NetworkHint,
312    ) -> (
313        ExternalBytesPort<NotMany>,
314        Stream<<Codec as Decoder>::Item, Self, Unbounded, TotalOrder, ExactlyOnce>,
315        ForwardHandle<'a, Stream<T, Self, Unbounded, TotalOrder, ExactlyOnce>>,
316    )
317    where
318        Self: Sized + NoTick,
319    {
320        let next_external_port_id = {
321            let mut flow_state = from.flow_state.borrow_mut();
322            let id = flow_state.next_external_out;
323            flow_state.next_external_out += 1;
324            id
325        };
326
327        let (fwd_ref, to_sink) =
328            self.forward_ref::<Stream<T, Self, Unbounded, TotalOrder, ExactlyOnce>>();
329        let mut flow_state_borrow = self.flow_state().borrow_mut();
330
331        flow_state_borrow.push_root(HydroRoot::SendExternal {
332            to_external_id: from.id,
333            to_key: next_external_port_id,
334            to_many: false,
335            unpaired: false,
336            serialize_fn: None,
337            instantiate_fn: DebugInstantiate::Building,
338            input: Box::new(to_sink.ir_node.into_inner()),
339            op_metadata: HydroIrOpMetadata::new(),
340        });
341
342        let raw_stream: Stream<
343            Result<<Codec as Decoder>::Item, <Codec as Decoder>::Error>,
344            Self,
345            Unbounded,
346            TotalOrder,
347            ExactlyOnce,
348        > = Stream::new(
349            self.clone(),
350            HydroNode::ExternalInput {
351                from_external_id: from.id,
352                from_key: next_external_port_id,
353                from_many: false,
354                codec_type: quote_type::<Codec>().into(),
355                port_hint,
356                instantiate_fn: DebugInstantiate::Building,
357                deserialize_fn: None,
358                metadata: self.new_node_metadata(Stream::<
359                    Result<<Codec as Decoder>::Item, <Codec as Decoder>::Error>,
360                    Self,
361                    Unbounded,
362                    TotalOrder,
363                    ExactlyOnce,
364                >::collection_kind()),
365            },
366        );
367
368        (
369            ExternalBytesPort {
370                process_id: from.id,
371                port_id: next_external_port_id,
372                _phantom: PhantomData,
373            },
374            raw_stream.flatten_ordered(),
375            fwd_ref,
376        )
377    }

380    fn bind_single_client_bincode<L, InT: DeserializeOwned, OutT: Serialize>(
381        &self,
382        from: &External<L>,
383    ) -> (
384        ExternalBincodeBidi<InT, OutT, NotMany>,
385        Stream<InT, Self, Unbounded, TotalOrder, ExactlyOnce>,
386        ForwardHandle<'a, Stream<OutT, Self, Unbounded, TotalOrder, ExactlyOnce>>,
387    )
388    where
389        Self: Sized + NoTick,
390    {
391        let next_external_port_id = {
392            let mut flow_state = from.flow_state.borrow_mut();
393            let id = flow_state.next_external_out;
394            flow_state.next_external_out += 1;
395            id
396        };
397
398        let (fwd_ref, to_sink) =
399            self.forward_ref::<Stream<OutT, Self, Unbounded, TotalOrder, ExactlyOnce>>();
400        let mut flow_state_borrow = self.flow_state().borrow_mut();
401
402        let root = get_this_crate();
403
404        let out_t_type = quote_type::<OutT>();
405        let ser_fn: syn::Expr = syn::parse_quote! {
406            ::#root::runtime_support::stageleft::runtime_support::fn1_type_hint::<#out_t_type, _>(
407                |b| #root::runtime_support::bincode::serialize(&b).unwrap().into()
408            )
409        };
410
411        flow_state_borrow.push_root(HydroRoot::SendExternal {
412            to_external_id: from.id,
413            to_key: next_external_port_id,
414            to_many: false,
415            unpaired: false,
416            serialize_fn: Some(ser_fn.into()),
417            instantiate_fn: DebugInstantiate::Building,
418            input: Box::new(to_sink.ir_node.into_inner()),
419            op_metadata: HydroIrOpMetadata::new(),
420        });
421
422        let in_t_type = quote_type::<InT>();
423
424        let deser_fn: syn::Expr = syn::parse_quote! {
425            |res| {
426                let b = res.unwrap();
427                #root::runtime_support::bincode::deserialize::<#in_t_type>(&b).unwrap()
428            }
429        };
430
431        let raw_stream: Stream<InT, Self, Unbounded, TotalOrder, ExactlyOnce> = Stream::new(
432            self.clone(),
433            HydroNode::ExternalInput {
434                from_external_id: from.id,
435                from_key: next_external_port_id,
436                from_many: false,
437                codec_type: quote_type::<LengthDelimitedCodec>().into(),
438                port_hint: NetworkHint::Auto,
439                instantiate_fn: DebugInstantiate::Building,
440                deserialize_fn: Some(deser_fn.into()),
441                metadata: self.new_node_metadata(Stream::<
442                    InT,
443                    Self,
444                    Unbounded,
445                    TotalOrder,
446                    ExactlyOnce,
447                >::collection_kind()),
448            },
449        );
450
451        (
452            ExternalBincodeBidi {
453                process_id: from.id,
454                port_id: next_external_port_id,
455                _phantom: PhantomData,
456            },
457            raw_stream,
458            fwd_ref,
459        )
460    }

697    /// Constructs a [`Singleton`] materialized at this location with the given static value.
698    ///
699    /// # Example
700    /// ```rust
701    /// # #[cfg(feature = "deploy")] {
702    /// # use hydro_lang::prelude::*;
703    /// # use futures::StreamExt;
704    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
705    /// let tick = process.tick();
706    /// let singleton = tick.singleton(q!(5));
707    /// # singleton.all_ticks()
708    /// # }, |mut stream| async move {
709    /// // 5
710    /// # assert_eq!(stream.next().await.unwrap(), 5);
711    /// # }));
712    /// # }
713    /// ```
714    fn singleton<T>(&self, e: impl QuotedWithContext<'a, T, Self>) -> Singleton<T, Self, Unbounded>
715    where
716        T: Clone,
717        Self: Sized,
718    {
719        // TODO(shadaj): we mark this as unbounded because we do not yet have a representation
720        // for bounded top-level singletons, and this is the only way to generate one
721
722        let e = e.splice_untyped_ctx(self);
723
724        Singleton::new(
725            self.clone(),
726            HydroNode::SingletonSource {
727                value: e.into(),
728                metadata: self
729                    .new_node_metadata(Singleton::<T, Self, Unbounded>::collection_kind()),
730            },
731        )
732    }

734    /// Generates a stream with values emitted at a fixed interval, with
735    /// each value being the current time (as an [`tokio::time::Instant`]).
736    ///
737    /// The clock source used is monotonic, so elements will be emitted in
738    /// increasing order.
739    ///
740    /// # Non-Determinism
741    /// Because this stream is generated by an OS timer, it will be
742    /// non-deterministic because each timestamp will be arbitrary.
743    fn source_interval(
744        &self,
745        interval: impl QuotedWithContext<'a, Duration, Self> + Copy + 'a,
746        _nondet: NonDet,
747    ) -> Stream<tokio::time::Instant, Self, Unbounded, TotalOrder, ExactlyOnce>
748    where
749        Self: Sized + NoTick,
750    {
751        self.source_stream(q!(tokio_stream::wrappers::IntervalStream::new(
752            tokio::time::interval(interval)
753        )))
754    }

756    /// Generates a stream with values emitted at a fixed interval (with an
757    /// initial delay), with each value being the current time
758    /// (as an [`tokio::time::Instant`]).
759    ///
760    /// The clock source used is monotonic, so elements will be emitted in
761    /// increasing order.
762    ///
763    /// # Non-Determinism
764    /// Because this stream is generated by an OS timer, it will be
765    /// non-deterministic because each timestamp will be arbitrary.
766    fn source_interval_delayed(
767        &self,
768        delay: impl QuotedWithContext<'a, Duration, Self> + Copy + 'a,
769        interval: impl QuotedWithContext<'a, Duration, Self> + Copy + 'a,
770        _nondet: NonDet,
771    ) -> Stream<tokio::time::Instant, Self, Unbounded, TotalOrder, ExactlyOnce>
772    where
773        Self: Sized + NoTick,
774    {
775        self.source_stream(q!(tokio_stream::wrappers::IntervalStream::new(
776            tokio::time::interval_at(tokio::time::Instant::now() + delay, interval)
777        )))
778    }

780    fn forward_ref<S>(&self) -> (ForwardHandle<'a, S>, S)
781    where
782        S: CycleCollection<'a, ForwardRef, Location = Self>,
783        Self: NoTick,
784    {
785        let next_id = self.flow_state().borrow_mut().next_cycle_id();
786        let ident = syn::Ident::new(&format!("cycle_{}", next_id), Span::call_site());
787
788        (
789            ForwardHandle {
790                completed: false,
791                ident: ident.clone(),
792                expected_location: Location::id(self),
793                _phantom: PhantomData,
794            },
795            S::create_source(ident, self.clone()),
796        )
797    }
