364pub(super) fn compile_sim(bin: String, trybuild: TrybuildConfig) -> Result<TempPath, ()> {
365    let mut command = Command::new("cargo");
366    command.current_dir(&trybuild.project_dir);
367    command.args(["rustc", "--locked"]);
368    command.args(["--example", "sim-dylib"]);
369    command.args(["--target-dir", trybuild.target_dir.to_str().unwrap()]);
370    if let Some(features) = &trybuild.features {
371        command.args(["--features", &features.join(",")]);
372    }
373    command.args(["--config", "build.incremental = false"]);
374    command.args(["--crate-type", "cdylib"]);
375    command.arg("--message-format=json-diagnostic-rendered-ansi");
376    command.env("STAGELEFT_TRYBUILD_BUILD_STAGED", "1");
377    command.env("TRYBUILD_LIB_NAME", &bin);
378
379    command.arg("--");
380
381    let is_fuzz = std::env::var("BOLERO_FUZZER").is_ok();
382    if is_fuzz {
383        command.env(
384            "RUSTFLAGS",
385            std::env::var("RUSTFLAGS_OUTER").unwrap_or_default() + " -C prefer-dynamic",
386        );
387    } else {
388        command.env(
389            "RUSTFLAGS",
390            std::env::var("RUSTFLAGS").unwrap_or_default() + " -C prefer-dynamic",
391        );
392    }
393
394    if cfg!(target_os = "linux") {
395        let debug_path = if let Ok(target) = std::env::var("CARGO_BUILD_TARGET")
396            && !is_fuzz
397        {
398            path!(trybuild.target_dir / target / "debug")
399        } else {
400            path!(trybuild.target_dir / "debug")
401        };
402
403        command.args([&format!(
404            "-Clink-arg=-Wl,-rpath,{}",
405            debug_path.to_str().unwrap()
406        )]);
407
408        if cfg!(target_env = "gnu") {
409            command.arg(
410                // https://github.com/rust-lang/rust/issues/91979
411                "-Clink-args=-Wl,-z,nodelete",
412            );
413        }
414    }
415
416    if let Ok(fuzzer) = std::env::var("BOLERO_FUZZER") {
417        command.env_remove("BOLERO_FUZZER");
418        command.env_remove("CARGO_BUILD_TARGET");
419
420        if fuzzer == "libfuzzer" {
421            #[cfg(target_os = "macos")]
422            {
423                command.args(["-Clink-arg=-undefined", "-Clink-arg=dynamic_lookup"]);
424            }
425
426            #[cfg(target_os = "linux")]
427            {
428                command.args(["-Clink-arg=-Wl,--unresolved-symbols=ignore-all"]);
429            }
430
431            command.args([
432                "-Cpasses=sancov-module",
433                "-Cllvm-args=-sanitizer-coverage-inline-8bit-counters",
434                "-Cllvm-args=-sanitizer-coverage-level=4",
435                "-Cllvm-args=-sanitizer-coverage-pc-table",
436                "-Cllvm-args=-sanitizer-coverage-trace-compares",
437            ]);
438        }
439    }
440
441    let mut spawned = command
442        .stdout(Stdio::piped())
443        .stdin(Stdio::null())
444        .spawn()
445        .unwrap();
446    let reader = std::io::BufReader::new(spawned.stdout.take().unwrap());
447
448    let mut out = Err(());
449    for message in cargo_metadata::Message::parse_stream(reader) {
450        match message.unwrap() {
451            cargo_metadata::Message::CompilerArtifact(artifact) => {
452                // unlike dylib, cdylib only exports the explicitly exported symbols
453                let is_output = artifact.target.is_example();
454
455                if is_output {
456                    use std::path::PathBuf;
457
458                    let path = artifact.filenames.first().unwrap();
459                    let path_buf: PathBuf = path.clone().into();
460                    out = Ok(path_buf);
461                }
462            }
463            cargo_metadata::Message::CompilerMessage(mut msg) => {
464                // Update the path displayed to enable clicking in IDE.
465                // TODO(mingwei): deduplicate code with hydro_deploy rust_crate/build.rs
466                if let Some(rendered) = msg.message.rendered.as_mut() {
467                    let file_names = msg
468                        .message
469                        .spans
470                        .iter()
471                        .map(|s| &s.file_name)
472                        .collect::<std::collections::BTreeSet<_>>();
473                    for file_name in file_names {
474                        *rendered = rendered.replace(
475                            file_name,
476                            &format!("(full path) {}/{file_name}", trybuild.project_dir.display()),
477                        )
478                    }
479                }
480                eprintln!("{}", msg.message);
481            }
482            cargo_metadata::Message::TextLine(line) => {
483                eprintln!("{}", line);
484            }
485            cargo_metadata::Message::BuildFinished(_) => {}
486            cargo_metadata::Message::BuildScriptExecuted(_) => {}
487            msg => panic!("Unexpected message type: {:?}", msg),
488        }
489    }
490
491    spawned.wait().unwrap();
492
493    let out_file = tempfile::NamedTempFile::new().unwrap().into_temp_path();
494    fs::copy(out.as_ref().unwrap(), &out_file).unwrap();
495    Ok(out_file)
496}

498pub(super) fn create_sim_graph_trybuild(
499    process_graphs: BTreeMap<LocationId, DfirGraph>,
500    cluster_graphs: BTreeMap<LocationId, DfirGraph>,
501    cluster_max_sizes: HashMap<LocationId, usize>,
502    process_tick_graphs: BTreeMap<LocationId, DfirGraph>,
503    cluster_tick_graphs: BTreeMap<LocationId, DfirGraph>,
504    extra_stmts_global: Vec<syn::Stmt>,
505    extra_stmts_cluster: BTreeMap<LocationId, Vec<syn::Stmt>>,
506) -> (String, TrybuildConfig) {
507    let source_dir = cargo::manifest_dir().unwrap();
508    let source_manifest = dependencies::get_manifest(&source_dir).unwrap();
509    let crate_name = &source_manifest.package.name.to_string().replace("-", "_");
510
511    let is_test = IS_TEST.load(std::sync::atomic::Ordering::Relaxed);
512
513    let generated_code = compile_sim_graph_trybuild(
514        process_graphs,
515        cluster_graphs,
516        cluster_max_sizes,
517        process_tick_graphs,
518        cluster_tick_graphs,
519        extra_stmts_global,
520        extra_stmts_cluster,
521        crate_name.clone(),
522        is_test,
523    );
524
525    let inlined_staged = if is_test {
526        let raw_toml_manifest = toml::from_str::<toml::Value>(
527            &fs::read_to_string(path!(source_dir / "Cargo.toml")).unwrap(),
528        )
529        .unwrap();
530
531        let maybe_custom_lib_path = raw_toml_manifest
532            .get("lib")
533            .and_then(|lib| lib.get("path"))
534            .and_then(|path| path.as_str());
535
536        let gen_staged = stageleft_tool::gen_staged_trybuild(
537            &maybe_custom_lib_path
538                .map(|s| path!(source_dir / s))
539                .unwrap_or_else(|| path!(source_dir / "src" / "lib.rs")),
540            &path!(source_dir / "Cargo.toml"),
541            crate_name.clone(),
542            Some("hydro___test".to_string()),
543        );
544
545        Some(prettyplease::unparse(&syn::parse_quote! {
546            #![allow(
547                unused,
548                ambiguous_glob_reexports,
549                clippy::suspicious_else_formatting,
550                unexpected_cfgs,
551                reason = "generated code"
552            )]
553
554            #gen_staged
555        }))
556    } else {
557        None
558    };
559
560    let source = prettyplease::unparse(&generated_code);
561
562    let hash = format!("{:X}", Sha256::digest(&source))
563        .chars()
564        .take(8)
565        .collect::<String>();
566
567    let bin_name = hash;
568
569    let (project_dir, target_dir, mut cur_bin_enabled_features) = create_trybuild().unwrap();
570
571    // TODO(shadaj): garbage collect this directory occasionally
572    fs::create_dir_all(path!(project_dir / "src")).unwrap();
573    fs::create_dir_all(path!(project_dir / "examples")).unwrap();
574
575    let out_path = path!(project_dir / "examples" / format!("{bin_name}.rs"));
576    {
577        let _concurrent_test_lock = CONCURRENT_TEST_LOCK.lock().unwrap();
578        write_atomic(source.as_ref(), &out_path).unwrap();
579    }
580
581    if let Some(inlined_staged) = inlined_staged {
582        let staged_path = path!(project_dir / "src" / "__staged.rs");
583        {
584            let _concurrent_test_lock = CONCURRENT_TEST_LOCK.lock().unwrap();
585            write_atomic(inlined_staged.as_bytes(), &staged_path).unwrap();
586        }
587    }
588
589    if is_test {
590        if cur_bin_enabled_features.is_none() {
591            cur_bin_enabled_features = Some(vec![]);
592        }
593
594        cur_bin_enabled_features
595            .as_mut()
596            .unwrap()
597            .push("hydro___test".to_string());
598    }
599
600    (
601        bin_name,
602        TrybuildConfig {
603            project_dir,
604            target_dir,
605            features: cur_bin_enabled_features,
606        },
607    )
608}

610#[expect(clippy::too_many_arguments, reason = "necessary for code generation")]
611fn compile_sim_graph_trybuild(
612    process_graphs: BTreeMap<LocationId, DfirGraph>,
613    cluster_graphs: BTreeMap<LocationId, DfirGraph>,
614    cluster_max_sizes: HashMap<LocationId, usize>,
615    process_tick_graphs: BTreeMap<LocationId, DfirGraph>,
616    cluster_tick_graphs: BTreeMap<LocationId, DfirGraph>,
617    extra_stmts_global: Vec<syn::Stmt>,
618    extra_stmts_cluster: BTreeMap<LocationId, Vec<syn::Stmt>>,
619    crate_name: String,
620    is_test: bool,
621) -> syn::File {
622    let mut diagnostics = Vec::new();
623
624    let mut dfir_into_code = |g: &DfirGraph| {
625        let mut dfir_expr: syn::Expr =
626            syn::parse2(g.as_code(&quote! { __root_dfir_rs }, true, quote!(), &mut diagnostics))
627                .unwrap();
628
629        if is_test {
630            UseTestModeStaged {
631                crate_name: crate_name.clone(),
632            }
633            .visit_expr_mut(&mut dfir_expr);
634        }
635
636        dfir_expr
637    };
638
639    let process_dfir_exprs = process_graphs
640        .into_iter()
641        .map(|(lid, g)| {
642            let dfir_expr = dfir_into_code(&g);
643            let ser_lid = serde_json::to_string(&lid).unwrap();
644            syn::parse_quote!((#ser_lid, None, #dfir_expr))
645        })
646        .collect::<Vec<syn::Expr>>();
647
648    let mut cluster_ticks_grouped_by_root = cluster_tick_graphs.into_iter().fold::<BTreeMap<
649        LocationId,
650        Vec<(LocationId, DfirGraph)>,
651    >, _>(
652        BTreeMap::new(),
653        |mut acc, (lid, g)| {
654            let root = lid.root();
655            acc.entry(root.clone()).or_default().push((lid, g));
656            acc
657        },
658    );
659
660    let cluster_dfir_stmts = cluster_graphs
661        .into_iter()
662        .map(|(lid, g)| {
663            let dfir_expr = dfir_into_code(&g);
664
665            let tick_dfir_stmts = cluster_ticks_grouped_by_root
666                .remove(&lid)
667                .unwrap_or_default()
668                .into_iter()
669                .map(|(tick_lid, tick_g)| {
670                    let tick_dfir_expr = dfir_into_code(&tick_g);
671                    let ser_tick_lid = serde_json::to_string(&tick_lid).unwrap();
672                    syn::parse_quote! {
673                        __tick_dfirs.push((
674                            #ser_tick_lid,
675                            Some(__current_cluster_id),
676                            #tick_dfir_expr
677                        ));
678                    }
679                })
680                .collect::<Vec<syn::Stmt>>();
681
682            let ser_lid = serde_json::to_string(&lid).unwrap();
683            let extra_stmts_per_cluster =
684                extra_stmts_cluster.get(&lid).cloned().unwrap_or_default();
685            let max_size = cluster_max_sizes.get(&lid).cloned().unwrap() as u32;
686
687            let cid = if let LocationId::Cluster(cid) = lid {
688                cid
689            } else {
690                unreachable!()
691            };
692
693            let self_id_ident = syn::Ident::new(
694                &format!("__hydro_lang_cluster_self_id_{}", cid),
695                Span::call_site(),
696            );
697
698            syn::parse_quote! {
699                for __current_cluster_id in 0..#max_size {
700                    __async_dfirs.push((
701                        #ser_lid,
702                        Some(__current_cluster_id),
703                        {
704                            #(#extra_stmts_per_cluster)*
705                            let #self_id_ident = &*Box::leak(Box::new(::hydro_lang::location::TaglessMemberId::from_raw_id(__current_cluster_id)));
706
707                            #(#tick_dfir_stmts)*
708
709                            #dfir_expr
710                        }
711                    ));
712                }
713            }
714        })
715        .collect::<Vec<syn::Stmt>>();
716
717    let process_tick_dfir_exprs = process_tick_graphs
718        .into_iter()
719        .map(|(lid, g)| {
720            let dfir_expr = dfir_into_code(&g);
721            let ser_lid = serde_json::to_string(&lid).unwrap();
722            syn::parse_quote!((#ser_lid, None, #dfir_expr))
723        })
724        .collect::<Vec<syn::Expr>>();
725
726    let trybuild_crate_name_ident = quote::format_ident!("{}_hydro_trybuild", crate_name);
727
728    let cluster_ids_stmts = cluster_max_sizes
729        .iter()
730        .map(|(lid, max_size)| {
731            let ident = syn::Ident::new(
732                &format!(
733                    "__hydro_lang_cluster_ids_{}",
734                    match lid {
735                        LocationId::Cluster(cid) => cid.to_string(),
736                        _ => panic!("Expected cluster location ID"),
737                    }
738                ),
739                Span::call_site(),
740            );
741
742            let elements = (0..*max_size as u32)
743                .map(|i| syn::parse_quote! { #i })
744                .collect::<Vec<syn::Expr>>();
745
746            syn::parse_quote! {
747                let #ident: &'static [::hydro_lang::location::TaglessMemberId] = Box::leak(Box::new([#(::hydro_lang::location::TaglessMemberId::from_raw_id(#elements)),*]));
748            }
749        })
750        .collect::<Vec<syn::Stmt>>();
751
752    let source_ast: syn::File = syn::parse_quote! {
753        use hydro_lang::prelude::*;
754        use hydro_lang::runtime_support::dfir_rs as __root_dfir_rs;
755        pub use #trybuild_crate_name_ident::__staged;
756
757        #[allow(unused)]
758        fn __hydro_runtime_core<'a>(
759            mut __hydro_external_out: ::std::collections::HashMap<usize, __root_dfir_rs::tokio::sync::mpsc::UnboundedSender<__root_dfir_rs::bytes::Bytes>>,
760            mut __hydro_external_in: ::std::collections::HashMap<usize, __root_dfir_rs::tokio_stream::wrappers::UnboundedReceiverStream<__root_dfir_rs::bytes::Bytes>>,
761            __println_handler: fn(::std::fmt::Arguments<'_>),
762            __eprintln_handler: fn(::std::fmt::Arguments<'_>),
763        ) -> (
764            Vec<(&'static str, Option<u32>, __root_dfir_rs::scheduled::graph::Dfir<'a>)>,
765            Vec<(&'static str, Option<u32>, __root_dfir_rs::scheduled::graph::Dfir<'a>)>,
766            hydro_lang::sim::runtime::Hooks<&'static str>,
767            hydro_lang::sim::runtime::InlineHooks<&'static str>,
768        ) {
769            macro_rules! println {
770                ($($arg:tt)*) => ({
771                    __println_handler(::std::format_args!($($arg)*));
772                })
773            }
774
775            macro_rules! eprintln {
776                ($($arg:tt)*) => ({
777                    __eprintln_handler(::std::format_args!($($arg)*));
778                })
779            }
780
781            // copy-pasted from std::dbg! so we can use the local eprintln! above
782            macro_rules! dbg {
783                // NOTE: We cannot use `concat!` to make a static string as a format argument
784                // of `eprintln!` because `file!` could contain a `{` or
785                // `$val` expression could be a block (`{ .. }`), in which case the `eprintln!`
786                // will be malformed.
787                () => {
788                    eprintln!("[{}:{}:{}]", ::std::file!(), ::std::line!(), ::std::column!())
789                };
790                ($val:expr $(,)?) => {
791                    // Use of `match` here is intentional because it affects the lifetimes
792                    // of temporaries - https://stackoverflow.com/a/48732525/1063961
793                    match $val {
794                        tmp => {
795                            eprintln!("[{}:{}:{}] {} = {:#?}",
796                                ::std::file!(),
797                                ::std::line!(),
798                                ::std::column!(),
799                                ::std::stringify!($val),
800                                // The `&T: Debug` check happens here (not in the format literal desugaring)
801                                // to avoid format literal related messages and suggestions.
802                                &&tmp as &dyn ::std::fmt::Debug,
803                            );
804                            tmp
805                        }
806                    }
807                };
808                ($($val:expr),+ $(,)?) => {
809                    ($(dbg!($val)),+,)
810                };
811            }
812
813            let mut __hydro_hooks: ::std::collections::HashMap<(&'static str, Option<u32>), ::std::vec::Vec<Box<dyn hydro_lang::sim::runtime::SimHook>>> = ::std::collections::HashMap::new();
814            let mut __hydro_inline_hooks: ::std::collections::HashMap<(&'static str, Option<u32>), ::std::vec::Vec<Box<dyn hydro_lang::sim::runtime::SimInlineHook>>> = ::std::collections::HashMap::new();
815            #(#extra_stmts_global)*
816            #(#cluster_ids_stmts)*
817
818            let mut __async_dfirs = vec![#(#process_dfir_exprs),*];
819            let mut __tick_dfirs = vec![#(#process_tick_dfir_exprs),*];
820            #(#cluster_dfir_stmts)*
821            (__async_dfirs, __tick_dfirs, __hydro_hooks, __hydro_inline_hooks)
822        }
823
824        #[unsafe(no_mangle)]
825        unsafe extern "Rust" fn __hydro_runtime(
826            should_color: bool,
827            __hydro_external_out: ::std::collections::HashMap<usize, __root_dfir_rs::tokio::sync::mpsc::UnboundedSender<__root_dfir_rs::bytes::Bytes>>,
828            __hydro_external_in: ::std::collections::HashMap<usize, __root_dfir_rs::tokio_stream::wrappers::UnboundedReceiverStream<__root_dfir_rs::bytes::Bytes>>,
829            __println_handler: fn(::std::fmt::Arguments<'_>),
830            __eprintln_handler: fn(::std::fmt::Arguments<'_>),
831        ) -> (
832            Vec<(&'static str, Option<u32>, __root_dfir_rs::scheduled::graph::Dfir<'static>)>,
833            Vec<(&'static str, Option<u32>, __root_dfir_rs::scheduled::graph::Dfir<'static>)>,
834            hydro_lang::sim::runtime::Hooks<&'static str>,
835            hydro_lang::sim::runtime::InlineHooks<&'static str>,
836        ) {
837            hydro_lang::runtime_support::colored::control::set_override(should_color);
838            __hydro_runtime_core(__hydro_external_out, __hydro_external_in, __println_handler, __eprintln_handler)
839        }
840    };
841    source_ast
