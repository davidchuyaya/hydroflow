19// same as the one in `hydro_std`, but internal use only
20fn track_membership<'a, C, L: Location<'a> + NoTick + NoAtomic>(
21    membership: KeyedStream<MemberId<C>, MembershipEvent, L, Unbounded>,
22) -> KeyedSingleton<MemberId<C>, (), L, Unbounded> {
23    membership
24        .fold(
25            q!(|| false),
26            q!(|present, event| {
27                match event {
28                    MembershipEvent::Joined => *present = true,
29                    MembershipEvent::Left => *present = false,
30                }
31            }),
32        )
33        .filter_map(q!(|v| if v { Some(()) } else { None }))
34}

36pub fn serialize_bincode_with_type(is_demux: bool, t_type: &syn::Type) -> syn::Expr {
37    let root = get_this_crate();
38
39    if is_demux {
40        parse_quote! {
41            ::#root::runtime_support::stageleft::runtime_support::fn1_type_hint::<(#root::location::MemberId<_>, #t_type), _>(
42                |(id, data)| {
43                    (id.raw_id, #root::runtime_support::bincode::serialize(&data).unwrap().into())
44                }
45            )
46        }
47    } else {
48        parse_quote! {
49            ::#root::runtime_support::stageleft::runtime_support::fn1_type_hint::<#t_type, _>(
50                |data| {
51                    #root::runtime_support::bincode::serialize(&data).unwrap().into()
52                }
53            )
54        }
55    }
56}

58fn serialize_bincode<T: Serialize>(is_demux: bool) -> syn::Expr {
59    serialize_bincode_with_type(is_demux, &quote_type::<T>())
60}

62pub fn deserialize_bincode_with_type(tagged: Option<&syn::Type>, t_type: &syn::Type) -> syn::Expr {
63    let root = get_this_crate();
64
65    if let Some(c_type) = tagged {
66        parse_quote! {
67            |res| {
68                let (id, b) = res.unwrap();
69                (#root::location::MemberId::<#c_type>::from_raw(id), #root::runtime_support::bincode::deserialize::<#t_type>(&b).unwrap())
70            }
71        }
72    } else {
73        parse_quote! {
74            |res| {
75                #root::runtime_support::bincode::deserialize::<#t_type>(&res.unwrap()).unwrap()
76            }
77        }
78    }
79}

81pub(crate) fn deserialize_bincode<T: DeserializeOwned>(tagged: Option<&syn::Type>) -> syn::Expr {
82    deserialize_bincode_with_type(tagged, &quote_type::<T>())
83}

85impl<'a, T, L, B: Boundedness, O, R> Stream<T, Cluster<'a, L>, B, O, R> {
86    pub fn send_bincode<L2>(
87        self,
88        other: &Process<'a, L2>,
89    ) -> KeyedStream<MemberId<L>, T, Process<'a, L2>, Unbounded, O, R>
90    where
91        T: Serialize + DeserializeOwned,
92    {
93        let serialize_pipeline = Some(serialize_bincode::<T>(false));
94
95        let deserialize_pipeline = Some(deserialize_bincode::<T>(Some(&quote_type::<L>())));
96
97        let raw_stream: Stream<(MemberId<L>, T), Process<'a, L2>, Unbounded, O, R> = Stream::new(
98            other.clone(),
99            HydroNode::Network {
100                serialize_fn: serialize_pipeline.map(|e| e.into()),
101                instantiate_fn: DebugInstantiate::Building,
102                deserialize_fn: deserialize_pipeline.map(|e| e.into()),
103                input: Box::new(self.ir_node.into_inner()),
104                metadata: other.new_node_metadata::<(MemberId<L>, T)>(),
105            },
106        );
107
108        raw_stream.into_keyed()
109    }
110
111    pub fn broadcast_bincode<L2: 'a>(
112        self,
113        other: &Cluster<'a, L2>,
114        nondet_membership: NonDet,
115    ) -> KeyedStream<MemberId<L>, T, Cluster<'a, L2>, Unbounded, O, R>
116    where
117        T: Clone + Serialize + DeserializeOwned,
118    {
119        let ids = track_membership(self.location.source_cluster_members(other));
120        let join_tick = self.location.tick();
121        let current_members = ids.snapshot(&join_tick, nondet_membership).keys();
122
123        current_members
124            .weaker_retries()
125            .assume_ordering::<TotalOrder>(
126                nondet!(/** we send to each member independently, order does not matter */),
127            )
128            .cross_product_nested_loop(
129                self.batch(&join_tick, nondet_membership)
130                    .assume_ordering::<TotalOrder>(
131                        nondet!(/** we weaken the ordering back later */),
132                    ),
133            )
134            .assume_ordering::<O>(nondet!(/** strictly weaker than TotalOrder */))
135            .all_ticks()
136            .demux_bincode(other)
137    }
138}

177impl<'a, T, L, B: Boundedness> Stream<T, Process<'a, L>, B, TotalOrder, ExactlyOnce> {
178    pub fn round_robin_bincode<L2: 'a>(
179        self,
180        other: &Cluster<'a, L2>,
181        nondet_membership: NonDet,
182    ) -> Stream<T, Cluster<'a, L2>, Unbounded, TotalOrder, ExactlyOnce>
183    where
184        T: Serialize + DeserializeOwned,
185    {
186        let ids = track_membership(self.location.source_cluster_members(other));
187        let join_tick = self.location.tick();
188        let current_members = ids
189            .snapshot(&join_tick, nondet_membership)
190            .keys()
191            .assume_ordering(
192                nondet!(/** safe to assume ordering because each output is independent */),
193            )
194            .collect_vec();
195
196        self.enumerate()
197            .batch(&join_tick, nondet_membership)
198            .cross_singleton(current_members)
199            .map(q!(|(data, members)| (
200                members[data.0 % members.len()],
201                data.1
202            )))
203            .all_ticks()
204            .demux_bincode(other)
205    }
206}

299    pub fn send_bincode_external<L2>(self, other: &External<L2>) -> ExternalBincodeStream<T>
300    where
301        T: Serialize + DeserializeOwned,
302    {
303        let serialize_pipeline = Some(serialize_bincode::<T>(false));
304
305        let mut flow_state_borrow = self.location.flow_state().borrow_mut();
306
307        let external_key = flow_state_borrow.next_external_out;
308        flow_state_borrow.next_external_out += 1;
309
310        let roots = flow_state_borrow.roots.as_mut().expect("Attempted to add a root to a flow that has already been finalized. No roots can be added after the flow has been compiled()");
311
312        roots.push(HydroRoot::SendExternal {
313            to_external_id: other.id,
314            to_key: external_key,
315            to_many: false,
316            serialize_fn: serialize_pipeline.map(|e| e.into()),
317            instantiate_fn: DebugInstantiate::Building,
318            input: Box::new(HydroNode::Unpersist {
319                inner: Box::new(self.ir_node.into_inner()),
320                metadata: self.location.new_node_metadata::<T>(),
321            }),
322            op_metadata: HydroIrOpMetadata::new(),
323        });
324
325        ExternalBincodeStream {
326            process_id: other.id,
327            port_id: external_key,
328            _phantom: PhantomData,
329        }
330    }
