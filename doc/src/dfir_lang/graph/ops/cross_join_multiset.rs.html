7/// > 2 input streams of type S and T, 1 output stream of type (S, T)
8///
9/// Forms the multiset cross-join (Cartesian product) of the (possibly duplicated) items in the input streams, returning all
10/// tupled pairs regardless of duplicates.
11///
12/// ```dfir
13/// source_iter(vec!["happy", "happy", "sad"]) -> [0]my_join;
14/// source_iter(vec!["dog", "cat", "cat"]) -> [1]my_join;
15/// my_join = cross_join_multiset() -> sort() -> assert_eq([
16///     ("happy", "cat"),
17///     ("happy", "cat"),
18///     ("happy", "cat"),
19///     ("happy", "cat"),
20///     ("happy", "dog"),
21///     ("happy", "dog"),
22///     ("sad", "cat"),
23///     ("sad", "cat"),
24///     ("sad", "dog"), ]);
25/// ```
26pub const CROSS_JOIN_MULTISET: OperatorConstraints = OperatorConstraints {
27    name: "cross_join_multiset",
28    categories: &[OperatorCategory::MultiIn],
29    hard_range_inn: &(2..=2),
30    soft_range_inn: &(2..=2),
31    hard_range_out: RANGE_1,
32    soft_range_out: RANGE_1,
33    num_args: 0,
34    persistence_args: &(0..=2),
35    type_args: &(0..=1),
36    is_external_input: false,
37    has_singleton_output: false,
38    flo_type: None,
39    ports_inn: Some(|| super::PortListSpec::Fixed(parse_quote! { 0, 1 })),
40    ports_out: None,
41    input_delaytype_fn: |_| None,
42    write_fn: |wc @ &WriteContextArgs {
43                   root,
44                   context,
45                   df_ident,
46                   op_span,
47                   work_fn_async,
48                   ident,
49                   inputs,
50                   ..
51               },
52               diagnostics| {
53        let lhs = &inputs[0];
54        let rhs = &inputs[1];
55
56        let [lhs_persistence, rhs_persistence] = wc.persistence_args_disallow_mutable(diagnostics);
57
58        let lhs_state = wc.make_ident("lhs_state");
59        let rhs_state = wc.make_ident("rhs_state");
60        let write_prologue = quote_spanned! {op_span=>
61            let #lhs_state = #df_ident.add_state(::std::cell::RefCell::new(::std::vec::Vec::new()));
62            let #rhs_state = #df_ident.add_state(::std::cell::RefCell::new(::std::vec::Vec::new()));
63        };
64
65        let lhs_write_prologue_after = wc
66            .persistence_as_state_lifespan(lhs_persistence)
67            .map(|lifespan| quote_spanned! {op_span=>
68                #df_ident.set_state_lifespan_hook(#lhs_state, #lifespan, move |rcell| { rcell.borrow_mut().clear(); });
69            }).unwrap_or_default();
70        let rhs_write_prologue_after = wc
71            .persistence_as_state_lifespan(rhs_persistence)
72            .map(|lifespan| quote_spanned! {op_span=>
73                #df_ident.set_state_lifespan_hook(#rhs_state, #lifespan, move |rcell| { rcell.borrow_mut().clear(); });
74            }).unwrap_or_default();
75        let write_prologue_after = quote_spanned! {op_span=>
76            #lhs_write_prologue_after
77            #rhs_write_prologue_after
78        };
79
80        let lhs_borrow = wc.make_ident("lhs_borrow");
81        let rhs_borrow = wc.make_ident("rhs_borrow");
82        let lhs_i = wc.make_ident("lhs_i");
83        let rhs_i = wc.make_ident("rhs_i");
84        let write_iterator = quote_spanned! {op_span=>
85            let (mut #lhs_borrow, mut #rhs_borrow) = unsafe {
86                // SAFETY: handle from `#df_ident.add_state(..)`.
87                (
88                    #context.state_ref_unchecked(#lhs_state).borrow_mut(),
89                    #context.state_ref_unchecked(#rhs_state).borrow_mut(),
90                )
91            };
92
93            let (#lhs_i, #rhs_i) = if #context.is_first_run_this_tick() {
94                (0, 0)
95            } else {
96                (#lhs_borrow.len(), #rhs_borrow.len())
97            };
98
99            #work_fn_async(#root::compiled::pull::ForEach::new(#lhs, |x| #lhs_borrow.push(x))).await;
100            #work_fn_async(#root::compiled::pull::ForEach::new(#rhs, |x| #rhs_borrow.push(x))).await;
101
102            let #ident = {
103                //       RHS
104                //   +-----+-----+
105                // L | Old | New |
106                // H +-----+-----+
107                // S | New | New |
108                //   +-----+-----+
109                #root::futures::stream::iter(
110                    #lhs_borrow
111                        .iter()
112                        .enumerate()
113                        .flat_map(|(i, lhs)| {
114                            let j = if i < #lhs_i { #rhs_i } else { 0 };
115                            #rhs_borrow[j..]
116                                .iter()
117                                .map(move |rhs| (::std::clone::Clone::clone(lhs), ::std::clone::Clone::clone(rhs)))
118                        })
119                )
120            };
121        };
122
123        let replay_code = matches!(
124            (lhs_persistence, rhs_persistence),
125            (Persistence::Static, Persistence::Static)
126        )
127        .then(|| {
128            quote_spanned! {op_span=>
129                // Reschedule the subgraph lazily to ensure replay on later ticks.
130                #context.schedule_subgraph(#context.current_subgraph(), false);
131            }
132        });
133        let write_iterator_after = quote_spanned! {op_span=>
134            #replay_code
135        };
136
137        // Ok(output)
138        Ok(OperatorWriteOutput {
139            write_prologue,
140            write_prologue_after,
141            write_iterator,
142            write_iterator_after,
143            ..Default::default()
144        })
145    },
146};
